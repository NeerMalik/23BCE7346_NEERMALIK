USE vitcampus;

SHOW TABLES FROM vitcampus;

SELECT * FROM WORKER;

SELECT FIRST_NAME, LAST_NAME FROM WORKER WHERE SALARY >= 100000 OR DEPARTMENT = 'Admin';

SELECT WORKER_ID, FIRST_NAME FROM WORKER WHERE WORKER_ID IN (1, 5, 6);

SELECT WORKER_ID, FIRST_NAME FROM WORKER WHERE WORKER_ID NOT IN (1, 5, 6);

SELECT WORKER_ID, LAST_NAME FROM WORKER WHERE SALARY BETWEEN 55000 AND 85000;

SELECT * FROM WORKER WHERE DEPARTMENT = 'HR' AND SALARY > 80000;

SELECT FIRST_NAME FROM WORKER WHERE FIRST_NAME LIKE '__i%';

SELECT DEPARTMENT, SUM(SALARY) AS total_salary FROM WORKER GROUP BY DEPARTMENT;

SELECT DEPARTMENT, COUNT(*) AS total_people FROM WORKER GROUP BY DEPARTMENT;

SELECT DEPARTMENT, AVG(SALARY) AS average_salary FROM WORKER GROUP BY DEPARTMENT;

SELECT DEPARTMENT, MIN(SALARY) AS floor_salary, MAX(SALARY) AS ceiling_salary FROM WORKER GROUP BY DEPARTMENT;

SELECT * FROM WORKER WHERE SALARY = (SELECT MIN(SALARY) FROM WORKER WHERE DEPARTMENT = 'Admin');

SELECT DEPARTMENT, SUM(SALARY) AS total_payroll FROM WORKER GROUP BY DEPARTMENT HAVING SUM(SALARY) > 300000;

SELECT WORKER_ID, FIRST_NAME, SALARY FROM WORKER ORDER BY SALARY DESC LIMIT 3;

SELECT SALARY FROM WORKER UNION SELECT SALARY FROM WORKER;

SELECT DISTINCT(DEPARTMENT) FROM WORKER;

SELECT FIRST_NAME, DEPARTMENT,
CASE DEPARTMENT
    WHEN 'HR' THEN 'Human Resources Team'
    WHEN 'Admin' THEN 'Administration Team'
    WHEN 'Account' THEN 'Finance Team'
    ELSE 'General Staff'
END AS department_description
FROM WORKER;

SELECT WORKER_ID, FIRST_NAME, SALARY,
CASE
    WHEN SALARY >= 300000 THEN 'Rich people'
    WHEN SALARY >= 100000 THEN 'Middle'
    ELSE 'Poor'
END AS Status
FROM WORKER;

CREATE VIEW v_worker_public_directory AS 
SELECT WORKER_ID, FIRST_NAME, LAST_NAME, DEPARTMENT FROM WORKER;

SELECT * FROM v_worker_public_directory;

CREATE OR REPLACE VIEW v_worker_public_directory AS 
SELECT DEPARTMENT, AVG(SALARY) AS average_department_salary FROM WORKER GROUP BY DEPARTMENT;
