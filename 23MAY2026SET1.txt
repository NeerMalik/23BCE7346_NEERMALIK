-- Create Employees Table
CREATE TABLE Employees (
emp_id INT PRIMARY KEY,
name VARCHAR(50) NOT NULL,

salary DECIMAL(10,2),
hire_date DATE,
FOREIGN KEY (dept_id) REFERENCES Departments(dept_id),
FOREIGN KEY (job_id) REFERENCES Jobs(job_id)
);

#1.[JOINS] List the name of each employee along with their job title and department name.
select name,job_title,dept_name from employees e1 join jobs j1 join departments d1 where e1.dept_id=d1.dept_id and e1.job_id=j1.job_id;
#2.[SUBQUERY] Find all employees whose salary is higher than the average salary of the 'IT' department.
select name from employees where salary>(select avg(salary) from employees e1 join departments d1 where e1.dept_id=d1.dept_id and d1.dept_name='IT'); 
#3.[GROUP BY] Calculate the total salary expenditure for each department name (not ID).
select dept_name,sum(salary) from employees e1 join departments d1 where e1.dept_id=d1.dept_id group by dept_name;
#4.[HAVING] Display departments that have more than 2 employees and an average salary above 70,000.
select dept_name from employees e1 join departments d1 where e1.dept_id=d1.dept_id group by dept_name having count(*)>2 and avg(salary)>70000;
#5.[SELF JOIN] Retrieve a list of employee names and their respective manager's names.
select e1.name,e2.name as manager_name from employees e1 join employees e2 where e2.emp_id=e1.manager_id;
#6.[ORDER BY] Find the top 3 highest-paid employees and list them in descending order.
select name,salary from employees order by salary desc limit 3; 
#7.[JOINS + WHERE] Find all Developers working in the 'San Francisco' location.
select name from employees e1 join departments d1 join jobs j1 where e1.dept_id=d1.dept_id and e1.job_id=j1.job_id and j1.job_title='Developer' and d1.location='San Francisco';
#8.[GROUP BY] Find the number of employees hired each year.
select year(hire_Date),count(*) from employees group by year(hire_Date);
#9.[HAVING] List job titles where the maximum salary of an employee in that role is less than 100,000.
select job_title from employees e1 join jobs j1 where e1.job_id=j1.job_id group by job_title having max(e1.salary)<100000;
#10.[SUBQUERY] Find the names of employees who work in the same department as 'David'.
select name from employees e1 join departments d1 where e1.dept_id=d1.dept_id and d1.dept_id=(select dept_id  from employees where name='David');
#11.[MULTI-TABLE JOIN] Show all department names even if they have no employees assigned to them.
select distinct(d1.dept_name) from departments d1 left join employees e1 on d1.dept_id = e1.dept_id;
#12.[COMPLEX FILTER] Find employees who were hired between 2018 and 2020 and earn more than 60,000.
select name, salary, hire_date from employees where year(hire_date) between 2018 and 2020 and salary > 60000;
#13.[CORRELATED SUBQUERY] Find employees who earn more than the average salary of their own department.
select name, salary from employees e1 where salary>(select avg(salary) from employees e2 where e1.dept_id = e2.dept_id);
#14.[JOINS] Find the job title and salary of the employee with ID 504.
select job_title, salary from employees e1 join jobs j1 on e1.job_id = j1.job_id where e1.emp_id = 504;
#16.[SUBQUERY] List departments that do not have any 'Analysts'.
select dept_name from departments where dept_id not in (select e1.dept_id from employees e1 join jobs j1 where e1.job_id = j1.job_id and j1.job_title = 'Analyst');
#17.[AGGREGATION] Find the difference between the highest and lowest salary in the company.
select max(salary) - min(salary) from employees;
#19.[HAVING] Identify managers (by ID) who manage more than 2 people.
select manager_id from employees where manager_id is not null group by manager_id having count(*) > 2;
#23.[MATH + GROUP BY] Calculate the 10% bonus amount for each employee and show the total bonus per department.
select d1.dept_name,SUM(e1.salary * 0.10) as total_bonus from employees e1 join departments d1 where e1.dept_id = d1.dept_id group by d1.dept_name;
#25.[JOINS] Find all employees who report to 'Charlie'.
select e1.name from employees e1 join employees e2 where e1.manager_id = e2.emp_id and e2.name = 'Charlie';
